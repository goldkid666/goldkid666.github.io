生成动态代理对象的工厂类

import org.springframework.stereotype.Component;
import java.lang.reflect.Proxy;

@Component
public class ProxyBeanFactory {
    //获取动态代理对象的方法
    public static Object getProxyBean(Object target){
        //获取目标对象类加载器，目标对象实现接口的数组
        ClassLoader classLoader = target.getClass().getClassLoader();
        Class<?>[] interfaces = target.getClass().getInterfaces();
        //new 一个 invocationHandler实现类，绑定上具体目标对象类 实现具体切面功能的添加
        MyInvocationHandler invocationHandler = new MyInvocationHandler(target);
        return Proxy.newProxyInstance(classLoader,interfaces,invocationHandler);
    }
}

插入日志功能的工具类

import org.springframework.stereotype.Component;

import java.util.Arrays;

@Component
public class MyLogging {
    public static void beforeMethod(String methodName,Object[] args){//jp是切入点 可获得数据
        System.out.println("执行" + methodName + "方法,"+"参数是：" + Arrays.toString(args) );
    }

    public static void afterMethod(String methodName){
        System.out.println("执行完" + methodName + "方法");
    }
}

具体的方法调用实现类 invocationHandler

invoke是具体实现方法，这里实现对应方法的调用，跟日志功能的添加

package com.atguigu.proxy;

import org.springframework.stereotype.Component;

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;

@Component
public class MyInvocationHandler implements InvocationHandler {
    private Object target;

    public MyInvocationHandler(Object target) {
        this.target = target;
    }

    /**
     *
     * @param proxy 目标对象
     * @param method 调用的相应方法的对象 通过此对象调用相应方法
     * @param args 调用的相应方法的参数
     * @return
     * @throws Throwable
     */
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        //获取methodname
        String methodName = method.getName();
        //调用之前调取日志
        MyLogging.beforeMethod(methodName,args);
        //对应方法调用
        Object invoke = method.invoke(target, args);
        //调用之后调取对应日志
        MyLogging.afterMethod(methodName);
        return invoke;
    }
}

import com.atguigu.aop.Calculator;
import com.atguigu.aop.impl.CalculatorImpl;
import com.atguigu.proxy.ProxyBeanFactory;
import org.junit.Test;

public class TestDymicProxy {
    @Test
    public void testDymicProxy(){
        //被代理对象(目标对象)
        CalculatorImpl calc = new CalculatorImpl();
        //通过动态代理对象工厂 获取指定的代理对象
        Calculator proxyBean = (Calculator) ProxyBeanFactory.getProxyBean(calc);
        //调取相应方法测试
        System.out.println(proxyBean.add(1, 2));
        System.out.println(proxyBean.multiply(2, 3));
    }
}
